import pandas as pd
import numpy as np
import torch
import torch.nn as nn
import torch.optim as optim
from sklearn.impute import SimpleImputer
from sklearn.preprocessing import StandardScaler
from torch.utils.data import DataLoader, TensorDataset

NUM_EPOCHS = 400
BATCH_SIZE = 32
LR_RATE = 3e-4


# Define device
device = torch.device('cuda' if torch.cuda.is_available() else 'cpu')

# Load and preprocess datasets
def preprocess(data):
    imputer = SimpleImputer(strategy='mean')
    scaler = StandardScaler()
    data_imputed = imputer.fit_transform(data)
    return scaler.fit_transform(data_imputed)

# Example: Load datasets (replace with real paths)
rna_path = "/content/drive/MyDrive/DATA____/simulation4/transcriptomics_data.csv"
proteomics_path = "/content/drive/MyDrive/DATA____/simulation4/proteomics_data.csv"
metabolomics_path = "/content/drive/MyDrive/DATA____/simulation4/metabolomics_data.csv"

transcriptomics = pd.read_csv(rna_path, index_col=0).T
proteomics = pd.read_csv(proteomics_path, index_col=0).T
metabolomics = pd.read_csv(metabolomics_path, index_col=0).T

# rna_preprocessed = preprocess(transcriptomics)
# proteomics_preprocessed = preprocess(proteomics)
# metabolomics_preprocessed = preprocess(metabolomics)
rna_preprocessed = transcriptomics
proteomics_preprocessed = proteomics
metabolomics_preprocessed = metabolomics

# Combine all modalities into a single dataset
multi_omics_data = np.hstack([rna_preprocessed, proteomics_preprocessed, metabolomics_preprocessed])
multi_omics_data = torch.tensor(multi_omics_data, dtype=torch.float32).to(device)

# Dataset and DataLoader
dataset = TensorDataset(multi_omics_data)
dataloader = DataLoader(dataset, batch_size=BATCH_SIZE, shuffle=True)

# MOVE VAE Model in PyTorch
class MOVEVAE(nn.Module):
    def __init__(self, input_dims, latent_dim):
        super(MOVEVAE, self).__init__()
        # Encoders for each modality
        self.encoders = nn.ModuleList([nn.Sequential(
            nn.Linear(dim, 128),
            nn.ReLU()
        ) for dim in input_dims])

        # Latent space
        self.latent_mean = nn.Linear(len(input_dims) * 128, latent_dim)
        self.latent_log_var = nn.Linear(len(input_dims) * 128, latent_dim)

        # Decoder
        self.decoder = nn.Sequential(
            nn.Linear(latent_dim, sum(input_dims)),
            nn.Sigmoid()
        )

    def encode(self, inputs):
        # Encode each modality and concatenate
        encoded = [encoder(inp) for encoder, inp in zip(self.encoders, inputs)]
        concatenated = torch.cat(encoded, dim=1)
        z_mean = self.latent_mean(concatenated)
        z_log_var = self.latent_log_var(concatenated)
        return z_mean, z_log_var

    def reparameterize(self, z_mean, z_log_var):
        epsilon = torch.randn_like(z_mean)
        return z_mean + torch.exp(0.5 * z_log_var) * epsilon

    def decode(self, z):
        return self.decoder(z)

    def forward(self, inputs):
        z_mean, z_log_var = self.encode(inputs)
        z = self.reparameterize(z_mean, z_log_var)
        reconstructed = self.decode(z)
        return reconstructed, z_mean, z_log_var

# Initialize model
input_dims = [rna_preprocessed.shape[1], proteomics_preprocessed.shape[1], metabolomics_preprocessed.shape[1]]
latent_dim = 10
move_vae = MOVEVAE(input_dims, latent_dim).to(device)



# Loss function
def move_loss(inputs, reconstructed, z_mean, z_log_var):
    reconstruction_loss = torch.mean((torch.cat(inputs, dim=1) - reconstructed) ** 2)
    kl_loss = -0.5 * torch.sum(1 + z_log_var - z_mean.pow(2) - z_log_var.exp())
    return reconstruction_loss + kl_loss

# Optimizer
optimizer = optim.Adam(move_vae.parameters(), lr=LR_RATE)

# Training Loop
epochs = NUM_EPOCHS
for epoch in range(epochs):
    total_loss = 0
    for batch in dataloader:
        # Split batch data into modalities
        batch_data = batch[0]
        rna_batch = batch_data[:, :input_dims[0]]
        proteomics_batch = batch_data[:, input_dims[0]:input_dims[0] + input_dims[1]]
        metabolomics_batch = batch_data[:, input_dims[0] + input_dims[1]:]

        inputs = [rna_batch, proteomics_batch, metabolomics_batch]

        # Forward pass
        reconstructed, z_mean, z_log_var = move_vae(inputs)

        # Compute loss
        loss = move_loss(inputs, reconstructed, z_mean, z_log_var)

        # Backward pass and optimization
        optimizer.zero_grad()
        loss.backward()
        optimizer.step()

        total_loss += loss.item()

    print(f"Epoch {epoch + 1}/{epochs}, Loss: {total_loss / len(dataloader)}")

# Save model
torch.save(move_vae.state_dict(), "move_vae.pth")
